\section{Expansion Laws}
\subsection{The Distributive Law and its Versions :}
\textbf{General Form : }{\boldmath$a(b+c) = ab + ac$}

{\footnotesize Example : $-3x(x-2) = (-3x)(x) - (-3x)(2) = -3x^2 + 6x$}\\
\textbf{Product of the sum of two terms :}{\boldmath$(a+b)(c+d) = (a+b)c + (a+b)d = ac + ad + bc + bd$}

{\footnotesize Example : $(4+2x)(x+5) = 4(x+5) + 2x(x+5) = 4x + 20 + 2x^2 + 10x = 2x^2 + 14x + 20$}\\
\textbf{Difference of two squares :} {\boldmath$(a+b)(a-b) = a^2 - b^2$}

{\footnotesize Example : $(2x+3)(2x-3) = (2x)^2 - (3)^2 = 4x^2 - 9$}

{\footnotesize Example : $(y^2 + \frac{3y}{2})(y^2 -\frac{3}{2}y) = y^4 - \frac{9}{4}y^2$}\\
\textbf{Perfect Squares Expansion :}

\textbf{a)}{\boldmath $(a+b)^2 = a^2 + 2ab + b^2$}

{\footnotesize Example : $(2x+3)^2 =(2x)^2 + 2(2x)(3)+(3)^2	= 4x^2 + 12x + 9$}

\textbf{b)}{\boldmath $(a-b)^2 = a^2 - 2ab + b^2$}

{\footnotesize Example : $(3x-4)^2 = (3x)^2 - 2(3x)(4)+(4)^2 = 9x^2 - 24x + 16$}

{\footnotesize Example : $(\frac{1}{2}x-2)^2 = (\frac{1}{2}x)^2 - 2(\frac{1}{2}x)(2) + (2)^2 = \frac{1}{4}x^2 - 2x + 4$}

\section{Further Expansion}

\begin{center}
{\boldmath$(a+b)(c+d) = a(c+d) + b(c+d) = ac + ad + bc + bd$}\\
{\boldmath$(a+b)(c+d+e)=a(c+d+e)+b(c+d+e)=ac+ad+ae+bc+bd+be$}
\end{center}
\subsection{Solved Examples}
\begin{enumerate}
	\begin{paracol}{2}
		\item[1)] Expand and simplify: $(x+1)(x^{2}+2x+3)$\\
		\textbf{Solution :}
		
		$(x+1)(x^{2}+2x+3) = x(x^{2}+2x+3) + 1(x^{2}+2x+3)\\= x^{3}+2x^{2}+3x+x^{2}+2x+3\\= x^{3}+3x^{2}+5x+3$
		
		Note: We always write terms in decreasing powers of $x$.
		\item[2)] Expand and simplify: $(2x+5)(3x-2)$\\
		\textbf{Solution :}
		
		$(2x+5)(3x-2) = 2x(3x-2) + 5(3x-2)\\= 6x^{2}-4x+15x-10\\= 6x^{2}+11x-10$
		\item[3)] Expand and simplify: $(2x-y+z)(2x-y+z)$\\
		\textbf{Solution :}
		
		$(2x-y+z)(2x-y+z) = 2x(2x-y+z)-y(2x-y+z)+z(2x-y+z)\\= 4x^{2}-2xy+2xz-2xy+y^{2}-yz+2xz-yz+z^{2}\\= 4x^{2}+y^{2}+z^{2}-4xy+4xz-2yz$
		\item[4)] Expand and simplify: $(a+b)(c+d)$\\
		\textbf{Solution :}
		
		$(a+b)(c+d) = a(c+d)+b(c+d)\\= ac+ad+bd+bd$
		\item[5)] Expand and simplify: $x(x-y)(x^{2}+y^{2}+xy)$\\
		\textbf{Solution :}
		
		$x(x-y)(x^{2}+y^{2}+xy)\\= x^{2}(x^{2}+y^{2}+xy)-xy(x^{2}+y^{2}+xy)\\= x^{4}+x^{2}y^{2}+x^{3}y-x^{3}y-xy^{3}-x^{3}y^{2}\\= x^{4}-xy^{3}$
		
		\switchcolumn
		
		\item[6)] Expand and simplify: $(x+\frac{1}{x})(x+\frac{1}{x}-2)$\\
		\textbf{Solution :}
		
		$(x+\frac{1}{x})(x+\frac{1}{x}-2) = x\left(x+\frac{1}{2}-2\right)+\frac{1}{x}\left(x+\frac{1}{3}-2\right)\\= x^{2}+\frac{1}{x^{2}}+2-2x-\frac{2}{x}$
		\item[7)] Expand and simplify: $(x-\frac{1}{x})(x-\frac{1}{x}-3)$\\
		\textbf{Solution :}
		
		$(x-\frac{1}{x})(x-\frac{1}{x}-3) = (x-\frac{1}{x})(x-\frac{1}{x}-3)\\ = x(x-\frac{1}{x}-3)-\frac{1}{x}(x-\frac{1}{x}-3)\\= x^2 -1 -3x-1+\frac{1}{x}^2 +\frac{3}{x}\\= x^2 + \frac{1}{x^2} - 2 - 3x + \frac{1}{x} + 3$
		\item[8)] Expand and simplify: $(a+x)(ax^{2}+1)$\\
		\textbf{Solution :}
		
		$(a+x)(ax^{2}+1)= a(a x^{2}+1) + x(ax)^{2}+1\\= a^2x^{2}+a+ax^{3}+x\\= a^2x^{2} + (ax^{2}+1)x + a$
		\item[9)] Expand and simplify: $(x+1+a)(x-1-a)$\\
		\textbf{Solution :}
		
		$(x+1+a)(x-1-a) = x(x-1-a)+1(x-1-a)+a(x-1-a)\\= x^2-x-ax+x-1-a+ax-a-a^2\\= x^2-1-2a-a^2$
		
		
		\item[10)] Expand and simplify: $(x+4)(x+5)(x+1)$\\
		\textbf{Solution :}
		
		$(x+4)(x+5)(x+1) = (x+4)\{x(x+1)+5(x+1)\}\\= (x+4)\{x^2+x+5x+5\}\\= (x+4)\{x^2+6x+5\}\\= (x^3+6x^2+5x)+(4x^2+24x+20)\\= x^3+10x^2+29x+20$

\subsection{Solved Examples}
\begin{enumerate}
	\begin{paracol}{2}
		\item[1)] Factorize $x^{2}+7x+12$

		\[
		\begin{aligned}
			x^{2}+3x+4x+12 &= x(x+3) + 4(x+3) \\
			&= (x+4)(x+3)
		\end{aligned}
		\]
		\item[2)] Factorize $x^{2}-x-12$
		
		Since the last term is negative, we split $12$ into two terms such that their difference is $-1$ and their product is $-12$.
		\[
		\begin{aligned}
			x^{2}-x-12 &= x^{2}-4x+3x-12 \\
			&= x(x-4) + 3(x-4) \\
			&= (x+3)(x-4)
		\end{aligned}
		\]
		\item[3)] Factorize $2x+11x+12$
		
		 Here we factorize $2 \times 12=24$ such that the sum $=11$ and product $=24$
		\[
		\begin{aligned}
			2x+11x+12 &= 2x^{2}+3x+8x+12 \\
			&= 2x(x+3) + 4(x+3) \\
			&= (2x+4)(x+3)
		\end{aligned}
		\]
		\item[4)] Factorize $28 x + 3x^{2} - x^{3} $
		$$
		\begin{aligned}
			28x + 3x^{2} - x^{3} &= -x\left(x^{2}-3 x-28\right) \\
			&= -x\left\{x^{2}-7 x+4 x-28\right\} \\
			&= -x\{x(x-7)+4(x-7)\} \\
			&= -x(x+4)(x+7)
		\end{aligned}
		$$

			&= -x^{2}\left\{x^{2}-9 x+3 x-27\right\} \\
			&= -x^{2}\{x(x-9)+3(x-9)\} \\
			&= -x^{2}(x+3)(x-9)
		\end{aligned}
		$$
		Note: We always remove a common factor first and change the sign so that first term of $x^{2}$ is positive
		\item[6)] Factorize $-16 x^{3} -4 x^{2}+30 x$
